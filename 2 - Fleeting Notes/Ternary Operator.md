```js
let isNightTime = true;  
  
if (isNightTime) {  
  console.log('Turn on the lights!');  
} else {  
  console.log('Turn off the lights!');  
}
```

=

```js
isNightTime ? console.log('Turn on the lights!') : console.log('Turn off the lights!');
```
